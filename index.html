<div id="calculator-container"></div>

<script>

    document.getElementById('calculator-container').innerHTML = `

        <div style="max-width: 400px; margin: auto; padding: 20px; border: 1px solid #ddd; border-radius: 8px;">

            <h2>최저시급 계산기</h2>

            <label for="hourlyWage">시간당 최저임금 (원):</label>

            <input type="number" id="hourlyWage" placeholder="예: 10030" style="width: 100%; padding: 10px; margin-top: 10px;">

            <label for="weeklyHours">주당 근무 시간:</label>

            <input type="number" id="weeklyHours" placeholder="예: 40" style="width: 100%; padding: 10px; margin-top: 10px;">

            <label><input type="checkbox" id="includeWeeklyAllowance"> 주휴수당 포함</label>

            <button onclick="calculateSalary()" style="width: 100%; padding: 10px; margin-top: 10px; background: #007bff; color: white; border: none; border-radius: 4px;">계산하기</button>

            <div style="margin-top: 20px; padding: 10px; background: #f9f9f9; border: 1px solid #ddd; border-radius: 4px;" id="result">결과가 여기에 표시됩니다.</div>

        </div>

        <script>

            function calculateSalary() {

                const hourlyWage = parseFloat(document.getElementById('hourlyWage').value);

                const weeklyHours = parseFloat(document.getElementById('weeklyHours').value);

                const includeWeeklyAllowance = document.getElementById('includeWeeklyAllowance').checked;

                if (isNaN(hourlyWage) || isNaN(weeklyHours)) {

                    alert('올바른 숫자를 입력해주세요.');

                    return;

                }

                const weeklyPay = hourlyWage * weeklyHours;

                const weeklyAllowance = includeWeeklyAllowance ? hourlyWage * (weeklyHours / 40) * 8 : 0;

                const totalWeeklyPay = weeklyPay + weeklyAllowance;

                const monthlyPay = totalWeeklyPay * 4.345;

                const yearlyPay = monthlyPay * 12;

                document.getElementById('result').innerHTML = `

                    <p><strong>주급:</strong> ${totalWeeklyPay.toLocaleString()} 원</p>

                    <p><strong>월급:</strong> ${monthlyPay.toLocaleString()} 원</p>

                    <p><strong>연봉:</strong> ${yearlyPay.toLocaleString()} 원</p>

                `;

            }

        </script>

    `;

</script>
